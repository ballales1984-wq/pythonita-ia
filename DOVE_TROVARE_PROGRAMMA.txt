🚀 COME AVVIARE IL PROGRAMMA

OPZIONE 1: DOPPIO CLICK (PIÙ SEMPLICE)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Apri Esplora File
2. Vai a: C:\Users\user\pythonita-ia\dist\
3. Trova: PythonitaIA.exe
4. Doppio click su PythonitaIA.exe
5. Il programma si avvia!


OPZIONE 2: DA TERMINALE
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Apri PowerShell nella cartella del progetto e digita:

.\dist\PythonitaIA.exe


OPZIONE 3: SCRIPT AVVIO VELOCE (BATCH)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Usa lo script già presente:

1. Vai a: C:\Users\user\pythonita-ia\
2. Doppio click su: AVVIA.bat

Questo script avvia automaticamente l'eseguibile!


OPZIONE 4: VERSIONE PYTHON (SVILUPPATORI)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

Se preferisci usare il codice sorgente Python:

python gui_robot_3d.py

Richiede:
- Python 3.8+
- Dependencies installate (pip install -r requirements.txt)
- spaCy modello italiano (python -m spacy download it_core_news_sm)

🐛 TROUBLESHOOTING

PROBLEMA: "Windows protected your PC"
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

SOLUZIONE:
1. Clicca "More info"
2. Clicca "Run anyway"

Questo è normale per exe non firmati digitalmente.


PROBLEMA: Exe non si avvia
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

SOLUZIONI:
1. Verifica che sia in: dist\PythonitaIA.exe
2. Verifica che file sia 120-130 MB
3. Prova a eseguire come amministratore
4. Controlla antivirus (potrebbe bloccarlo)
5. Usa versione Python: python gui_robot_3d.py


PROBLEMA: Manca spaCy modello
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

SOLUZIONE (solo per versione Python):
python -m spacy download it_core_news_sm

L'exe ha già tutto incluso!


PROBLEMA: Errore librerie mancanti
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

SOLUZIONE:
L'exe è standalone, non servono librerie!
Se usi Python: pip install -r requirements.txt
